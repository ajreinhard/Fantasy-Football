Poll = Class.create();
Poll.barImagePath = 'http://g.espncdn.com/s/ffllm/14/images/pollBar.gif';
Poll.theMonths = new Array("Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec");
Poll.today = new Date();

Poll.prototype = {
	
	pollCode: -1,
	isTeamVoting: false,
	pollResponse: null,
	isPassive: true,
	isAllowedToVote: true,
	options: [],
	votes: [],
	maxBarWidth: 0,
	responseUrl: '',
	selectedOption: null,
		
	barWidth: 0,
	pollBodyElementName: '',
	pollOptionsElementName: '',
	pollSelOptionElementName: '',
	
	initialize: function(args) {	
	
		this.pollCode = args.pollCode;
		this.isTeamVoting = args.isTeamVoting;
		this.isPassive = args.isPassive;
		this.isAllowedToVote = args.isAllowedToVote;
		this.maxBarWidth = args.maxBarWidth;
		this.responseUrl = args.responseUrl;
		this.options = args.options;
		this.votes = args.votes;
		this.pollResponse = args.pollResponse;

		if (this.pollResponse != null && this.pollResponse.vote < 0) {
			this.pollResponse = null;
		}

		this.pollBodyElementName = 'poll' + this.pollCode + '_body';	
		this.pollOptionsElementName = 'poll' + this.pollCode + '_options';
		this.pollSelOptionElementName = 'poll' + this.pollCode + '_selOpt';	
		
		this.render();		
	},

	render:function() {
		if (this.pollResponse != null || !this.isAllowedToVote) { //  || this.isPassive

			this.renderResultsView();
		} else {		
			this.renderVoteView();			
		}
	},
	
	vote: function() {

		if (this.selectedOption > -1) {
		
			this.pollResponse = new PollResponse(this.selectedOption);
			v = this.pollResponse.vote
			if (this.isAllowedToVote) {
				this.votes[this.selectedOption]++;
				this.render();
				if (this.isPassive) { 
					var passiveResURL = String(this.responseUrl).replace("PASSIVEVOTE", v);
					window.location.href = passiveResURL;
				} else { window.location.href = this.responseUrl + v; }
			}
		} 
		
	},
	
	renderResultsView: function() {
		var newHtml = '';
		var totalVotes = 0;
		
		for(i = 0; i < this.votes.length; i++) {
			totalVotes += this.votes[i];
		}
		
		Element.update(this.pollBodyElementName, '');
	
		for(var ct = 0; ct < this.options.length; ct++) {
			var opt = this.options[ct];
			if (this.votes[ct] == 0) {
				var barWidth = 1;
				var votePct = 0;
			} else {
				var barWidth = Math.max(Math.round(this.maxBarWidth * (this.votes[ct] / totalVotes)), 1);
				var votePct = Math.round((100 / totalVotes) * this.votes[ct]);
			}
			var plural = (this.votes[ct] != 1 ? 's' : '');
			newHtml += '<div style="display:block;margin-top:8px;">';
			if (this.votes[ct] > 0) {
				newHtml += '<img src="' + Poll.barImagePath + '" height="10" width="' + barWidth + '" border="0" /> ';
			}
			newHtml += votePct + '%<br />' + opt;
			if (this.pollResponse != null && this.pollResponse.vote == ct) {
				newHtml += '*';
			}
			newHtml += ' <nobr>(' + this.votes[ct] + ' vote' + plural + ')</nobr></div>';
		}
		
		newHtml += '<div style="display:block;margin-top:16px;">'
		if (this.pollResponse != null) {
			newHtml += '<em>You';			
			newHtml += ' voted on ' + this.pollResponse.voteDate + '</em><br>';
		} 
		
		newHtml += 'Total Votes: ' + totalVotes + '&nbsp;&nbsp;(One per ';
		if(this.isTeamVoting) { 
			newHtml += 'Team' 
		} else { 
			newHtml += 'Owner' 
		}; 
		newHtml += ')</div>';	
		preloadImage = new Image(1, 10); 
		preloadImage.src=Poll.barImagePath; 
		new Insertion.Bottom($(this.pollBodyElementName), newHtml);
	},
	
	renderVoteView: function() {
		var _this = this;
		Element.update(this.pollBodyElementName, '');
		
		for(var ct = 0; ct < this.options.length; ct++) {
			radioId = this.pollOptionsElementName + ct;
		
			newHtml = '<input name="' + this.pollOptionsElementName + '" id="' + radioId + '" type="radio" value="' + ct + '" >' + this.options[ct] + '<br />\n';	
			new Insertion.Bottom($(this.pollBodyElementName), newHtml);		
			
			$(radioId).onclick = function () {
				_this.selectedOption = this.value;
			}
		}
		new Insertion.Bottom($(this.pollBodyElementName), '<br />');
		
		var voteButton = document.createElement('INPUT');
		voteButton.value = 'Vote!';
		voteButton.type = 'button';
		voteButton.className = 'sidebar';
		voteButton.onclick = function (event) {   
			_this.vote();
		};		
		$(this.pollBodyElementName).appendChild(voteButton);
			
		newHtml = '<br />(One vote per ';		
		if(this.isTeamVoting) { newHtml += 'Team' } else { newHtml += 'Owner' }; 
		newHtml += ')';
		
		new Insertion.Bottom($(this.pollBodyElementName), newHtml);	
	}
}
